DEVICES:
    D1, D2, D3, D4 := DTYPE;
    SW_SET, SW_CLEAR := SWITCH;
    Clock := CLOCK;
END;

    D1.QBAR, D2.QBAR, D3.QBAR, D4.QBAR => D1.DATA, D2.DATA, D3.DATA, D4.DATA;
    SW_SET => D1.SET, D2.SET, D3.SET, D4.SET;
    SW_CLEAR => D1.CLEAR, D2.CLEAR, D3.CLEAR, D4.CLEAR; 
    Clock, D1.Q, D2.Q, D3.Q => D1.CLK, D2.CLK, D3.CLK, D4.CLK; 
END;

MONITORS:
    Clock, D1.Q, D2.Q, D3.Q, D4.Q;
END;
